{# ── Single Item Fetch ── {% single post from collection.post %} looks up the item whose slug matches the URL parameter (e.g. /post/my-first-post → slug = "my-first-post"). Inside the block, "post" is the variable name — access fields with post.title, post.body, post.featured_image, etc. {% else %} handles the case when no item matches the slug (404-like). #} {% single post from collection.post %} {# Dynamic meta — set and description from the item's fields #} {{ meta.title }}{{ post.title }}{{ /meta.title }} {{ meta.description }}{{ post.excerpt }}{{ /meta.description }} {# Featured image #} {% if post.featured_image %} <img src="{{ post.featured_image | image }}" alt="{{ post.title }}" style="max-width: 100%; height: auto; margin-bottom: 24px;"> {% endif %} {# Title #} <h1>{{ post.title }}</h1> {# Meta info — date and author #} <div style="color: var(--color-muted); font-size: 14px; margin-bottom: 24px;"> {% if post.published_at %} <time>{{ post.published_at | date }}</time> {% endif %} {% if post.author %} <span> · {{ post.author }}</span> {% endif %} </div> {# Body — richtext field rendered as HTML #} <div style="line-height: 1.7;"> {{ post.body | raw }} </div> {# Back link #} <div style="margin-top: 40px; padding-top: 24px; border-top: 1px solid #e5e7eb;"> <a href="/blog" style="color: var(--color-accent); text-decoration: none;">← Back to all posts</a> </div> {% else %} {# Not found — displayed when no item matches the URL slug #} {{ meta.title }}Post Not Found{{ /meta.title }} <h1>Post Not Found</h1> <p style="color: var(--color-muted);">The post you're looking for doesn't exist or has been removed.</p> <a href="/blog" style="color: var(--color-accent); text-decoration: none;">← Back to blog</a> {% endsingle %} </main> {% include 'footer' %}
