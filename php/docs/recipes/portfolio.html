Portfolio

Showcase your work with image galleries, case studies, client info, and a filterable project grid. This recipe is ideal for designers, developers, photographers, and agencies.

Collections Gallery Folders Filtering

1. Collection setup

Go to Admin → Collections → New Collection and create:

SettingValue
NameProjects
Slugproject
URL pattern/work/{slug}

Add these fields to the schema:

Field nameTypeNotes
titleTextProject name
excerptTextareaShort summary for the grid card
clientTextClient name, shown on the project page
yearTextYear completed, e.g. 2024
cover_imageImageGrid thumbnail and page hero
galleryGalleryMultiple project images — rendered with a gallery loop
bodyRichtextCase study / project description
live_urlLinkLive project URL (optional)
tagsSelectInternal tags, e.g. Web, Mobile, Print

2. Folder setup

Create one flat folder for project categories, attached to the project collection:

5. work.html — Filterable project grid

The grid shows all projects with category filter links. Clicking a category appends ?category=slug and the filteredby:category modifier does the rest.

<outpost-include partial="head" />
<outpost-include partial="nav" />

<main>
  <header class="page-header">
    <h1>Work</h1>
  </header>

  <!-- Category filter nav — folder loop -->
  <nav class="filter-nav">
    <a href="/work" class="filter-link" data-label="">All</a>
    <outpost-each folder="category">
      <a href="/work?category=<span data-outpost="slug">slug</span>"
         class="filter-link"
         data-label="<span data-outpost="slug">slug</span>">
        <span data-outpost="name">Category Name</span>
      </a>
    </outpost-each>
  </nav>

  <!-- Project grid — filtered by ?category= query param -->
  <div class="project-grid">
    <outpost-each collection="project" filteredby="category">
      <a href="<span data-outpost="url">#</span>" class="project-card">
        <outpost-if field="cover_image" exists>
          <div class="project-card-image">
            <img data-outpost="cover_image" data-type="image"
                 src="default.jpg"
                 alt="<span data-outpost="title">Project</span>"
                 loading="lazy">
          </div>
        </outpost-if>
        <div class="project-card-info">
          <h2><span data-outpost="title">Project Title</span></h2>
          <outpost-if field="excerpt" exists>
            <p><span data-outpost="excerpt">Short description</span></p>
          </outpost-if>
          <outpost-if field="year" exists>
            <span class="project-year"><span data-outpost="year">2024</span></span>
          </outpost-if>
        </div>
      </a>
    </outpost-each>
  </div>
</main>

<!-- Highlight active category filter -->
<script>
  const params = new URLSearchParams(location.search);
  const active = params.get('category') || '';
  document.querySelectorAll('.filter-link').forEach(a => {
    if (a.dataset.label === active) a.classList.add('active');
  });
</script>

6. project.html — Single project page

The single project page renders the cover image, gallery, client info, richtext case study, and a live URL link.

<outpost-include partial="head" />
<outpost-include partial="nav" />

<main>
  <outpost-each collection="project" single>

    <!-- Hero cover image -->
    <outpost-if field="cover_image" exists>
      <div class="project-hero">
        <img data-outpost="cover_image" data-type="image" src="default.jpg" alt="<span data-outpost="title">Project</span>">
      </div>
    </outpost-if>

    <div class="project-layout">
      <!-- Sidebar — project metadata -->
      <aside class="project-meta">
        <h1><span data-outpost="title">Project Title</span></h1>
        <outpost-if field="client" exists>
          <dl>
            <dt>Client</dt>
            <dd><span data-outpost="client">Client Name</span></dd>
          </dl>
        </outpost-if>
        <outpost-if field="year" exists>
          <dl>
            <dt>Year</dt>
            <dd><span data-outpost="year">2024</span></dd>
          </dl>
        </outpost-if>
        <outpost-if field="live_url" exists>
          <a data-outpost="live_url" data-type="link"
             href="#"
             class="btn-live"
             target="_blank"
             rel="noopener">
            View Live Site
          </a>
        </outpost-if>
        <a href="/work" class="back-link">← All Work</a>
      </aside>

      <!-- Main column — case study body -->
      <div class="project-body">
        <div data-outpost="body" data-type="richtext">Project description goes here.</div>
      </div>
    </div>

    <!-- Gallery — renders each image in the gallery field -->
    <outpost-if field="gallery" exists>
      <section class="project-gallery">
        <h2>Gallery</h2>
        <div class="gallery-grid">
          <outpost-each repeat="gallery">
            <a href="<span data-outpost="url">#</span>" class="gallery-item">
              <img src="<span data-outpost="url">default.jpg</span>"
                   alt="<span data-outpost="alt">Gallery image</span>"
                   loading="lazy">
            </a>
          </outpost-each>
        </div>
      </section>
    </outpost-if>

  </outpost-each>
  <div class="not-found">
    <h1>Project not found</h1>
    <a href="/work">Back to Work</a>
  </div>
</main>

7. Gallery grid CSS

A minimal CSS grid for the photo gallery. Adjust grid-template-columns to change the column count.

/* gallery-grid: 3-column responsive photo grid */
.gallery-grid {
  display: grid;
  grid-template-columns: repeat(3, 1fr);
  gap: 8px;
}

.gallery-item {
  display: block;
  aspect-ratio: 4 / 3;
  overflow: hidden;
}

.gallery-item img {
  width: 100%;
  height: 100%;
  object-fit: cover;
  transition: transform 0.3s ease;
}

.gallery-item:hover img {
  transform: scale(1.03);
}

@media (max-width: 768px) {
  .gallery-grid {
    grid-template-columns: repeat(2, 1fr);
  }
}

@media (max-width: 480px) {
  .gallery-grid {
    grid-template-columns: 1fr;
  }
}
Tip For a lightbox effect, add a data attribute to each <a> tag and use a lightweight JS library such as GLightbox or PhotoSwipe to open full-size images on click. No Outpost-specific changes needed.

8. Filtered view using filteredby

The filteredby="category" attribute on the collection loop reads the ?category= query parameter automatically. When no parameter is present, all projects are shown. When a category slug is supplied, only projects tagged with that label are returned.

<!-- Show all projects — no filter param in URL -->
<outpost-each collection="project" filteredby="category">
  ...
</outpost-each>

<!-- Visiting /work?category=branding shows only Branding projects -->
<!-- Visiting /work shows all projects -->
Note Only one filteredby attribute can be used per loop. If you need to filter by multiple folders simultaneously, fetch from the Content API and implement filtering client-side.

9. Homepage featured projects

The homepage typically shows a selection of your best work — not the full grid. Use limit="6" and optionally a custom sort to feature specific projects.

<outpost-include partial="head" />
<outpost-include partial="nav" />

<main>
  <!-- Hero — tagline from global field -->
  <section class="home-hero">
    <h1><span data-outpost="site_name" data-scope="global">Site Name</span></h1>
    <outpost-if field="site_tagline" data-scope="global" exists>
      <p class="tagline"><span data-outpost="site_tagline" data-scope="global">Your tagline here</span></p>
    </outpost-if>
    <a href="/work" class="cta-link">View all work →</a>
  </section>

  <!-- Featured projects — most recent 6 -->

    <div class="project-grid">
      <outpost-each collection="project" limit="6">
        <a href="<span data-outpost="url">#</span>" class="project-card">
          <outpost-if field="cover_image" exists>
            <div class="project-card-image">
              <img data-outpost="cover_image" data-type="image"
                   src="default.jpg"

                   loading="lazy">
            </div>
          </outpost-if>
          <span class="project-title"><span data-outpost="title">Project Title</span></span>
        </a>
      </outpost-each>
    </div>
  </section>
</main>

10. About page with a Repeater field

If your about page needs a list of services, team members, or process steps, add a Repeater field to a page and loop over it in your template. This keeps the structure editable in the admin without requiring custom fields for every row.

<!-- Example: a "services" repeater with fields: name, description -->
<outpost-if field="services" exists>
  <section class="services">
    <h2>What we do</h2>
    <ul>
      <outpost-each repeat="services">
        <li>
          <strong><span data-outpost="name">Service Name</span></strong>
          <p><span data-outpost="description">Service description</span></p>
        </li>
      </outpost-each>
    </ul>
  </section>
</outpost-if>

11. Contact page

A simple contact page outputting global fields for email, Instagram, and Behance. Link to a form tool or embed a third-party form (Tally, Typeform, etc.) as an iframe or script embed in the richtext field.

</main>
